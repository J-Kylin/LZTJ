    let currentIndex = 0;
    let scores = { shenyi:0, ducheng:0, lihan:0, jiangfeng:0, laoyan:0 };
    let chaosCount = 0;
    let shuffledQuestions = [];

    function shuffleArray(arr) {
        for (let i = arr.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
        return arr;
    }

    function initShuffledQuestions() {
        shuffledQuestions = RAW_QUESTIONS.map(q => {
            let shuffledOpts = shuffleArray([...q.options]);
            return {
                text: q.text,
                options: shuffledOpts
            };
        });
    }

    function updateProgress() {
        document.getElementById('currentQNum').innerText = currentIndex + 1;
        document.getElementById('chaosCounter').innerHTML = `🎭 乱答次数: ${chaosCount}`;
    }

    function showFinalResult() {
        if (chaosCount >= 15) {
            const isShenyi = Math.random() < 0.5;
            const youthKey = isShenyi ? 'shenyi_youth' : 'ducheng_youth';
            const youth = RESULTS[youthKey];
            renderResultPage(youth.name, youth.title, youth.desc, youth.link);
        } else {
            let maxRole = null;
            let maxScore = -1;
            for (let [role, score] of Object.entries(scores)) {
                if (score > maxScore) {
                    maxScore = score;
                    maxRole = role;
                }
            }
            if (!maxRole) maxRole = "shenyi";
            const normalResult = RESULTS[maxRole];
            renderResultPage(normalResult.name, normalResult.title, normalResult.desc, normalResult.link);
        }
    }

    function renderResultPage(name, title, desc, link) {
        const hasLink = link && link.trim() !== "" && link !== "#";
        let jumpBtnHtml = "";
        if (hasLink) {
            jumpBtnHtml = `<button class="jump-btn" id="jumpToLinkBtn">🚀 前往专属页面 →</button>`;
        }
        const resultHtml = `
            <div class="result-area">
                <div class="result-icon">🔮</div>
                <div class="result-name">${name}</div>
                <div class="result-sub">${title}</div>
                <div class="result-desc">${desc}</div>
                <div class="btn-group">
                    <button class="restart-btn" id="restartTestBtn">🔄 重新测试</button>
                    ${jumpBtnHtml}
                </div>
            </div>
        `;
        document.getElementById('resultPanel').innerHTML = resultHtml;
        document.getElementById('questionPanel').classList.add('hidden');
        document.getElementById('resultPanel').classList.remove('hidden');
        
        const restartBtn = document.getElementById('restartTestBtn');
        if (restartBtn) restartBtn.addEventListener('click', () => fadeOutThen(() => resetAndRestart()));
        
        const jumpBtn = document.getElementById('jumpToLinkBtn');
        if (jumpBtn && hasLink) {
            jumpBtn.addEventListener('click', () => {
                document.body.classList.add('fade-out');
                setTimeout(() => window.location.href = link, 300);
            });
        }
    }

    function resetAndRestart() {
        currentIndex = 0;
        scores = { shenyi:0, ducheng:0, lihan:0, jiangfeng:0, laoyan:0 };
        chaosCount = 0;
        initShuffledQuestions();
        updateProgress();
        renderCurrentQuestion();
    }

    function fadeOutThen(callback) {
        document.body.classList.add('fade-out');
        setTimeout(() => {
            callback();
            document.body.classList.remove('fade-out');
        }, 280);
    }

    function handleSkip() {
        chaosCount++;
        updateProgress();
        if (currentIndex + 1 < shuffledQuestions.length) {
            currentIndex++;
            updateProgress();
            renderCurrentQuestion();
        } else {
            fadeOutThen(() => {
                showFinalResult();
            });
        }
    }

    function handleOptionClick(opt) {
        if (opt.role) {
            scores[opt.role] = (scores[opt.role] || 0) + 1;
        }
        if (currentIndex + 1 < shuffledQuestions.length) {
            currentIndex++;
            updateProgress();
            renderCurrentQuestion();
        } else {
            fadeOutThen(() => {
                showFinalResult();
            });
        }
    }

    function renderCurrentQuestion() {
        const q = shuffledQuestions[currentIndex];
        const optionsHtml = q.options.map(opt => {
            return `<button class="option-btn" data-opt='${JSON.stringify(opt)}'>${opt.text}</button>`;
        }).join('');
        const skipHtml = `<div class="skip-btn-container"><button class="skip-btn" id="skipBtn">🙅 我就不答！（点击跳过题目）</button></div>`;
        const html = `
            <div class="question-text">${q.text}</div>
            <div class="options">
                ${optionsHtml}
            </div>
            ${skipHtml}
        `;
        document.getElementById('questionPanel').innerHTML = html;
        document.getElementById('questionPanel').classList.remove('hidden');
        document.getElementById('resultPanel').classList.add('hidden');
        
        document.querySelectorAll('#questionPanel .option-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const optData = JSON.parse(btn.dataset.opt);
                handleOptionClick(optData);
            });
        });
        
        const skipBtn = document.getElementById('skipBtn');
        if (skipBtn) {
            skipBtn.addEventListener('click', handleSkip);
        }
    }

    function initTest() {
        initShuffledQuestions();
        currentIndex = 0;
        scores = { shenyi:0, ducheng:0, lihan:0, jiangfeng:0, laoyan:0 };
        chaosCount = 0;
        updateProgress();
        renderCurrentQuestion();
    }

    initTest();
